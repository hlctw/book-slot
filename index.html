// 1. 在 HTML head 中加入 Supabase

<script src="https://unpkg.com/@supabase/supabase-js@2"></script>


// 2. 初始化 Supabase 客戶端

const supabaseUrl = 'YOUR_SUPABASE_URL'

const supabaseKey = 'YOUR_SUPABASE_ANON_KEY'

const supabase = supabase.createClient(supabaseUrl, supabaseKey)


// 3. 載入課程（替換 loadAllCourses 函數）

async function loadAllCourses() {

    setStatus('🔄 載入課程中...', 'loading');

    

    try {

        const { data, error } = await supabase

            .from('sessions')

            .select('*')

            .order('startDate', { ascending: true });

            

        if (error) throw error;

        

        allCourses = data || [];

        displayCourses(allCourses);

        setStatus('✅ 載入完成', 'success');

        document.getElementById('courseCounter').textContent = 

            '共找到 ' + allCourses.length + ' 個課程';

            

    } catch (error) {

        console.error('❌ 載入失敗:', error);

        setStatus('❌ 載入失敗：' + error.message, 'error');

    }

}
